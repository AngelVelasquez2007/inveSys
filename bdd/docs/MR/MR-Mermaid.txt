    categories ||--o{ products : classifies
    products ||--o{ product_suppliers : has
    suppliers ||--o{ product_suppliers : supplies

    roles {
        BIGINT role_id PK
        VARCHAR role_name
        TEXT description
        TIMESTAMPTZ created_at
    }

    users {
        BIGINT user_id PK
        BIGINT role_id FK
        VARCHAR first_name
        VARCHAR last_name
        VARCHAR email
        VARCHAR password_hash
        BOOLEAN active
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }

    categories {
        BIGINT category_id PK
        VARCHAR category_name
        TEXT description
        TIMESTAMPTZ created_at
    }

    products {
        BIGINT product_id PK
        VARCHAR sku
        VARCHAR product_name
        BIGINT category_id FK
        NUMERIC price
        INTEGER stock_quantity
        INTEGER minimum_stock
        BOOLEAN active
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }

    suppliers {
        BIGINT supplier_id PK
        VARCHAR nit
        VARCHAR supplier_name
        VARCHAR email
        VARCHAR city
        BOOLEAN active
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }

    supplier_phones {
        BIGINT supplier_phone_id PK
        BIGINT supplier_id FK
        VARCHAR phone
    }

    product_suppliers {
        BIGINT product_id PK, FK
        BIGINT supplier_id PK, FK
        NUMERIC cost
        INTEGER delivery_days
        BOOLEAN is_primary
        TIMESTAMPTZ created_at
    }

    customers {
        BIGINT customer_id PK
        VARCHAR document_number
        VARCHAR first_name
        VARCHAR last_name
        VARCHAR email
        VARCHAR city
        BOOLEAN active
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }

    customer_phones {
        BIGINT customer_phone_id PK
        BIGINT customer_id FK
        VARCHAR phone
    }

    sales_orders {
        BIGINT order_id PK
        BIGINT customer_id FK
        BIGINT user_id FK
        VARCHAR status
        TIMESTAMPTZ order_date
        NUMERIC total_amount
    }

    sales_order_items {
        BIGINT order_item_id PK
        BIGINT order_id FK
        BIGINT product_id FK
        INTEGER quantity
        NUMERIC unit_price
        NUMERIC line_total
    }

    inventory_movements {
        BIGINT movement_id PK
        BIGINT product_id FK
        VARCHAR movement_type
        INTEGER quantity
        VARCHAR reason
        TIMESTAMPTZ created_at
    }

    audit_logs {
        BIGINT audit_id PK
        BIGINT user_id FK
        VARCHAR action
        VARCHAR table_name
        TEXT record_id
        JSONB old_values
        JSONB new_values
        INET ip_address
        TIMESTAMPTZ created_at
    }
