    roles {
        BIGSERIAL role_id PK "NOT NULL"
        VARCHAR_60 role_name "NOT NULL UNIQUE"
        TEXT description "NULL"
        TIMESTAMPTZ created_at "NOT NULL DEFAULT now()"
    }

    users {
        BIGSERIAL user_id PK "NOT NULL"
        BIGINT role_id FK "NOT NULL REFERENCES roles(role_id)"
        VARCHAR_80 first_name "NOT NULL"
        VARCHAR_80 last_name "NOT NULL"
        VARCHAR_160 email "NOT NULL UNIQUE"
        VARCHAR_255 password_hash "NOT NULL"
        BOOLEAN active "NOT NULL DEFAULT true"
        TIMESTAMPTZ created_at "NOT NULL DEFAULT now()"
        TIMESTAMPTZ updated_at "NOT NULL DEFAULT now()"
    }

    categories {
        BIGSERIAL category_id PK "NOT NULL"
        VARCHAR_80 category_name "NOT NULL UNIQUE"
        TEXT description "NULL"
        TIMESTAMPTZ created_at "NOT NULL DEFAULT now()"
    }

    products {
        BIGSERIAL product_id PK "NOT NULL"
        VARCHAR_30 sku "NOT NULL UNIQUE"
        VARCHAR_120 product_name "NOT NULL"
        BIGINT category_id FK "NOT NULL REFERENCES categories(category_id)"
        NUMERIC_12_2 price "NOT NULL CHECK price > 0"
        INTEGER stock_quantity "NOT NULL DEFAULT 0 CHECK stock_quantity >= 0"
        INTEGER minimum_stock "NOT NULL DEFAULT 0 CHECK minimum_stock >= 0"
        BOOLEAN active "NOT NULL DEFAULT true"
        TIMESTAMPTZ created_at "NOT NULL DEFAULT now()"
        TIMESTAMPTZ updated_at "NOT NULL DEFAULT now()"
    }

    suppliers {
        BIGSERIAL supplier_id PK "NOT NULL"
        VARCHAR_30 nit "NOT NULL UNIQUE"
        VARCHAR_120 supplier_name "NOT NULL"
        VARCHAR_160 email "NOT NULL UNIQUE"
        VARCHAR_80 city "NOT NULL"
        BOOLEAN active "NOT NULL DEFAULT true"
        TIMESTAMPTZ created_at "NOT NULL DEFAULT now()"
        TIMESTAMPTZ updated_at "NOT NULL DEFAULT now()"
    }

    supplier_phones {
        BIGSERIAL supplier_phone_id PK "NOT NULL"
        BIGINT supplier_id FK "NOT NULL REFERENCES suppliers(supplier_id) ON DELETE CASCADE"
        VARCHAR_30 phone "NOT NULL"
    }

    product_suppliers {
        BIGINT product_id PK, FK "NOT NULL REFERENCES products(product_id) ON DELETE CASCADE"
        BIGINT supplier_id PK, FK "NOT NULL REFERENCES suppliers(supplier_id) ON DELETE RESTRICT"
        NUMERIC_12_2 cost "NOT NULL CHECK cost > 0"
        INTEGER delivery_days "NOT NULL DEFAULT 3 CHECK delivery_days > 0"
        BOOLEAN is_primary "NOT NULL DEFAULT false"
        TIMESTAMPTZ created_at "NOT NULL DEFAULT now()"
    }

    customers {
        BIGSERIAL customer_id PK "NOT NULL"
        VARCHAR_30 document_number "NOT NULL UNIQUE"
        VARCHAR_80 first_name "NOT NULL"
        VARCHAR_80 last_name "NOT NULL"
        VARCHAR_160 email "NOT NULL UNIQUE"
        VARCHAR_80 city "NOT NULL"
        BOOLEAN active "NOT NULL DEFAULT true"
        TIMESTAMPTZ created_at "NOT NULL DEFAULT now()"
        TIMESTAMPTZ updated_at "NOT NULL DEFAULT now()"
    }

    customer_phones {
        BIGSERIAL customer_phone_id PK "NOT NULL"
        BIGINT customer_id FK "NOT NULL REFERENCES customers(customer_id) ON DELETE CASCADE"
        VARCHAR_30 phone "NOT NULL"
    }

    sales_orders {
        BIGSERIAL order_id PK "NOT NULL"
        BIGINT customer_id FK "NOT NULL REFERENCES customers(customer_id)"
        BIGINT user_id FK "NULL REFERENCES users(user_id)"
        VARCHAR_20 status "NOT NULL DEFAULT pending CHECK status IN pending paid cancelled delivered"
        TIMESTAMPTZ order_date "NOT NULL DEFAULT now()"
        NUMERIC_14_2 total_amount "NOT NULL DEFAULT 0 CHECK total_amount >= 0"
    }

    sales_order_items {
        BIGSERIAL order_item_id PK "NOT NULL"
        BIGINT order_id FK "NOT NULL REFERENCES sales_orders(order_id) ON DELETE CASCADE"
        BIGINT product_id FK "NOT NULL REFERENCES products(product_id)"
        INTEGER quantity "NOT NULL CHECK quantity > 0"
        NUMERIC_12_2 unit_price "NOT NULL CHECK unit_price > 0"
        NUMERIC_14_2 line_total "GENERATED quantity * unit_price"
    }

    inventory_movements {
        BIGSERIAL movement_id PK "NOT NULL"
        BIGINT product_id FK "NOT NULL REFERENCES products(product_id)"
        VARCHAR_20 movement_type "NOT NULL CHECK movement_type IN in out adjustment"
        INTEGER quantity "NOT NULL CHECK quantity > 0"
        VARCHAR_160 reason "NOT NULL"
        TIMESTAMPTZ created_at "NOT NULL DEFAULT now()"
    }

    audit_logs {
        BIGSERIAL audit_id PK "NOT NULL"
        BIGINT user_id FK "NULL REFERENCES users(user_id)"
        VARCHAR_10 action "NOT NULL CHECK action IN INSERT UPDATE DELETE"
        VARCHAR_80 table_name "NOT NULL"
        TEXT record_id "NULL"
        JSONB old_values "NULL"
        JSONB new_values "NULL"
        INET ip_address "DEFAULT inet_client_addr()"
        TIMESTAMPTZ created_at "NOT NULL DEFAULT now()"
    }
